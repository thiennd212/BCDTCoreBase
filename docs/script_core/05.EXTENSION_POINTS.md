## 1. Authentication Provider

### 1.1. Interface

```csharp
public interface IAuthenticationProvider
{
    string ProviderType { get; }  // "BuiltIn", "SSO", "LDAP"
    bool IsEnabled { get; }
    int Priority { get; }         // Lower = higher priority
    
    Task<AuthResult> AuthenticateAsync(AuthRequest request);
    Task<bool> ValidateTokenAsync(string token);
    Task<AuthResult> RefreshTokenAsync(string refreshToken);
    Task LogoutAsync(string userId, string? sessionId = null);
    Task<bool> CanHandleAsync(AuthRequest request);
}

public interface IExternalAuthProvider : IAuthenticationProvider
{
    string GetAuthorizationUrl(string redirectUri, string state);
    Task<AuthResult> HandleCallbackAsync(string code, string state);
    Task<ExternalUserInfo> GetUserInfoAsync(string accessToken);
}

public interface IDirectoryAuthProvider : IAuthenticationProvider
{
    Task<bool> ValidateCredentialsAsync(string username, string password);
    Task<DirectoryUserInfo> GetUserFromDirectoryAsync(string username);
    Task SyncUsersAsync();
}
```

### 1.2. MVP: BuiltIn Provider

```csharp
public class BuiltInAuthProvider : IAuthenticationProvider
{
    public string ProviderType => "BuiltIn";
    public bool IsEnabled => true;
    public int Priority => 100;

    public async Task<AuthResult> AuthenticateAsync(AuthRequest request)
    {
        var user = await _userRepository.GetByUsernameAsync(request.Username);
        
        if (user == null || !_passwordHasher.Verify(request.Password, user.PasswordHash))
        {
            return AuthResult.Failed("INVALID_CREDENTIALS");
        }
        
        var accessToken = _jwtService.GenerateAccessToken(user);
        var refreshToken = _jwtService.GenerateRefreshToken();
        
        return new AuthResult
        {
            Success = true,
            AccessToken = accessToken,
            RefreshToken = refreshToken,
            User = MapToUserInfo(user)
        };
    }
}
```

### 2.1. Interface

```csharp
public interface ITwoFactorProvider

{

    Task<TwoFactorResult> ValidateAsync(string sessionId, string code);
    Task<IEnumerable<string>> GetAvailableProvidersAsync();

```

### 2.2. Future: TOTP Provider (Skeleton)

```csharp
public class TotpProvider : ITwoFactorProvider
{
    public string ProviderType => "TOTP";
    public bool IsEnabled => _settings.TotpEnabled;

    public async Task<TwoFactorSetupResult> SetupAsync(int userId)
    {
        var secret = GenerateSecret();
        var backupCodes = GenerateBackupCodes(10);
        
        await _repository.SaveSecretAsync(userId, "TOTP", secret, backupCodes);
        
        var qrUri = GenerateQrCodeUri(secret, user.Email, "BCDT");
        
        return new TwoFactorSetupResult
        {
            Success = true,
            Secret = secret,
            QrCodeUri = qrUri,
            BackupCodes = backupCodes
        };
    }

    public async Task<bool> ValidateAsync(int userId, string code)
    {
        var secret = await _repository.GetSecretAsync(userId, "TOTP");
        var totp = new Totp(Base32Encoding.ToBytes(secret));
        return totp.VerifyTotp(code, out _, new VerificationWindow(1, 1));
    }
}
```

## 3. Digital Signature

### 3.1. Interface

```csharp
public interface ISignatureProvider
{
    string ProviderType { get; }  // "Audit", "Simple", "VGCA"
    bool IsEnabled { get; }
    bool RequiresHardwareToken { get; }
    
    Task<SignatureResult> SignAsync(SignatureRequest request);
    Task<VerificationResult> VerifyAsync(string documentId, string signature);
    Task<byte[]> GetSignedDocumentAsync(string documentId);
}

public interface IVgcaSignatureProvider : ISignatureProvider
{
    Task<CertificateInfo> GetCertificateInfoAsync();
    Task<SignatureResult> SignWithTimestampAsync(SignatureRequest request);
}
```

### 3.2. MVP: Audit-based Signature

```csharp
public class AuditBasedSignatureProvider : ISignatureProvider
{
    public string ProviderType => "Audit";
    public bool IsEnabled => true;
    public bool RequiresHardwareToken => false;

    public async Task<SignatureResult> SignAsync(SignatureRequest request)
    {
        var documentHash = await _hashService.ComputeDocumentHashAsync(request.DocumentId);
        
        var signature = new DocumentSignature
        {
            Id = Guid.NewGuid().ToString("N"),
            DocumentId = request.DocumentId,
            SignerId = request.SignerId,
            SignerRole = request.SignerRole,
            SignatureType = SignatureType.Approval,
            DocumentHashAtSigning = Convert.ToBase64String(documentHash),
            SignedAt = DateTime.UtcNow
        };
        
        await _repository.SaveSignatureAsync(signature);
        await _auditService.LogAsync(AuditAction.DocumentSigned, ...);
        
        return SignatureResult.Success(signature.Id);
    }

    public async Task<VerificationResult> VerifyAsync(string documentId, string signatureId)
    {
        var signature = await _repository.GetSignatureAsync(signatureId);
        var currentHash = await _hashService.ComputeDocumentHashAsync(documentId);
        
        var isValid = signature.DocumentHashAtSigning == Convert.ToBase64String(currentHash);
        
        return new VerificationResult
        {
            IsValid = isValid,
            Message = isValid ? "Document unchanged" : "Document modified after signing"
        };
    }
}
```

### 3.3. Future: VGCA Provider (Skeleton)

```csharp
public class VgcaSignatureProvider : IVgcaSignatureProvider
{
    public string ProviderType => "VGCA";
    public bool IsEnabled => _settings.VgcaEnabled;
    public bool RequiresHardwareToken => true;

    public async Task<SignatureResult> SignWithTimestampAsync(SignatureRequest request)
    {
        // 1. Get certificate from USB Token (client-side)
        var certificate = await GetCertificateFromTokenAsync();
        
        // 2. Create PKCS#7 signature
        var signature = await CreatePkcs7SignatureAsync(request.DocumentHash, certificate);
        
        // 3. Get timestamp from VGCA TSA
        var timestampToken = await _vgcaClient.GetTimestampAsync(signature);
        
        // 4. Embed timestamp
        var signedData = await EmbedTimestampAsync(signature, timestampToken);
        
        return SignatureResult.Success(signedData);
    }
}
```

## 4. DI Registration

```csharp
public static class ServiceCollectionExtensions
{
    public static IServiceCollection AddBcdtExtensibleServices(
        this IServiceCollection services,
        IConfiguration configuration)
    {
        // Authentication
        services.AddScoped<IAuthenticationService, AuthenticationService>();
        services.AddScoped<IAuthenticationProvider, BuiltInAuthProvider>();
        services.AddScoped<IAuthenticationProvider, SsoAuthProvider>();      // Disabled by config
        services.AddScoped<IAuthenticationProvider, LdapAuthProvider>();     // Disabled by config

        // Two-Factor
        services.AddScoped<ITwoFactorService, TwoFactorService>();
        services.AddScoped<ITwoFactorProvider, TotpProvider>();              // Disabled by config
        services.AddScoped<ITwoFactorProvider, SmsCodeProvider>();           // Disabled by config

        // Signature
        services.AddScoped<ISignatureService, SignatureService>();
        services.AddScoped<ISignatureProvider, AuditBasedSignatureProvider>();
        services.AddScoped<ISignatureProvider, VgcaSignatureProvider>();     // Disabled by config

        return services;
    }
}
```
